#include <stdio.h>
int multiply(int *arr, int n) {
    int product = 1;
    for(int i = 0; i < n; i++) {
        product = product * arr[i];  
    }
    return product;
}
int main() {
    int arr[5], result;

    printf("Enter 5 numbers: ");
    for(int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    result = multiply(arr, 5); 
 printf("Multiplication = %d\n", result);

    return 0;
}
