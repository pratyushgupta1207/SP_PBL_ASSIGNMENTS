#include <stdio.h>

int main() {
    FILE *fptr;
    char data[100];

    
    fptr = fopen("myfile.txt", "w"); 
    if (fptr == NULL) {
        printf("Error creating file!");
        return 1;
    }

    printf("Enter some text to write in file: ");
    gets(data);  
    fprintf(fptr, "%s", data);
    fclose(fptr);

    printf("\nData written successfully!\n");

    
    fptr = fopen("myfile.txt", "r");  
    printf("\nReading from file:\n");
    while (fgets(data, sizeof(data), fptr)) {
        printf("%s", data);
    }
    fclose(fptr);

    
    if (remove("myfile.txt") == 0)
        printf("\n\nFile deleted successfully!\n");
    else
        printf("\n\nError deleting file!\n");

    return 0;
}S
