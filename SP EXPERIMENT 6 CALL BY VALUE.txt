#include <stdio.h>


void swapByValue(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    printf("Inside swapByValue - a: %d, b: %d\n", a, b);
}

int main() {
    int x = 10, y = 20;

    printf("Before swapByValue - x: %d, y: %d\n", x, y);
    swapByValue(x, y);
    printf("After swapByValue - x: %d, y: %d\n", x, y);

    return 0;
}
